// MEDIDAS DAX PARA INVESTMENT DASHBOARD
// Copiar y pegar en PowerBI Desktop → Modelado → Nueva medida
// =================================

// 🏦 MEDIDAS PRINCIPALES
// Total AUM
Total AUM = SUM('Funds Overview Demo'[TotalAssets])

// Average Return
Average Return = AVERAGE('Funds Overview Demo'[YTDReturn])

// Count Active Funds
Active Funds = COUNTROWS(FILTER('Funds Overview Demo', 'Funds Overview Demo'[Status] = "Active"))

// Best Performer
Top Fund Return = MAX('Funds Overview Demo'[YTDReturn])

// Worst Performer  
Worst Fund Return = MIN('Funds Overview Demo'[YTDReturn])

// 📊 MEDIDAS AVANZADAS
// Total AUM in Billions
AUM Billions = [Total AUM] / 1000000000

// Performance Score
Performance Score = 
SWITCH(
    TRUE(),
    [Average Return] >= 0.15, "Excellent",
    [Average Return] >= 0.08, "Good", 
    [Average Return] >= 0.03, "Average",
    "Poor"
)

// Risk Weighted Return
Risk Weighted Return = 
SUMX(
    'Funds Overview Demo',
    'Funds Overview Demo'[YTDReturn] * 'Funds Overview Demo'[TotalAssets]
) / [Total AUM]

// High Risk Funds Count
High Risk Funds = COUNTROWS(FILTER('Funds Overview Demo', 'Funds Overview Demo'[RiskLevel] = "High"))

// Percentage High Risk
% High Risk = DIVIDE([High Risk Funds], [Active Funds], 0)

// 💹 MEDIDAS DE DISTRIBUCIÓN
// Equity Allocation
Equity AUM = 
CALCULATE(
    [Total AUM],
    'Funds Overview Demo'[AssetClass] = "Equity"
)

// Fixed Income Allocation  
Fixed Income AUM = 
CALCULATE(
    [Total AUM],
    'Funds Overview Demo'[AssetClass] = "Fixed Income"
)

// Alternative Investments
Alternative AUM = 
CALCULATE(
    [Total AUM],
    OR(
        'Funds Overview Demo'[AssetClass] = "Real Estate",
        'Funds Overview Demo'[AssetClass] = "Commodity"
    )
)

// 📈 MEDIDAS DE PERFORMANCE
// Positive Return Funds
Positive Return Funds = COUNTROWS(FILTER('Funds Overview Demo', 'Funds Overview Demo'[YTDReturn] > 0))

// Percentage Positive
% Positive Returns = DIVIDE([Positive Return Funds], [Active Funds], 0)

// Outperforming Funds (>10%)
Outperforming Funds = COUNTROWS(FILTER('Funds Overview Demo', 'Funds Overview Demo'[YTDReturn] > 0.10))

// Average Assets per Fund
Avg Assets per Fund = DIVIDE([Total AUM], [Active Funds], 0)

// 🎯 MEDIDAS PARA KPIS
// YTD Return Range
Return Range = [Top Fund Return] - [Worst Fund Return]

// Portfolio Volatility (simplified)
Portfolio Volatility = 
VAR ReturnStdDev = 
    SQRT(
        SUMX(
            'Funds Overview Demo',
            POWER('Funds Overview Demo'[YTDReturn] - [Average Return], 2)
        ) / [Active Funds]
    )
RETURN ReturnStdDev

// Asset Concentration Risk
Largest Fund Weight = 
DIVIDE(
    MAX('Funds Overview Demo'[TotalAssets]),
    [Total AUM],
    0
)

// 📊 MEDIDAS PARA GRÁFICOS
// Return Category Count
Excellent Performers = COUNTROWS(FILTER('Funds Overview Demo', 'Funds Overview Demo'[YTDReturn] >= 0.15))
Good Performers = COUNTROWS(FILTER('Funds Overview Demo', AND('Funds Overview Demo'[YTDReturn] >= 0.08, 'Funds Overview Demo'[YTDReturn] < 0.15)))
Average Performers = COUNTROWS(FILTER('Funds Overview Demo', AND('Funds Overview Demo'[YTDReturn] >= 0.03, 'Funds Overview Demo'[YTDReturn] < 0.08)))
Poor Performers = COUNTROWS(FILTER('Funds Overview Demo', 'Funds Overview Demo'[YTDReturn] < 0.03))

// 🔄 MEDIDAS DINÁMICAS
// Selected Fund AUM  
Selected Fund AUM = 
IF(
    HASONEVALUE('Funds Overview Demo'[FundName]),
    SUM('Funds Overview Demo'[TotalAssets]),
    BLANK()
)

// Selected Fund Return
Selected Fund Return = 
IF(
    HASONEVALUE('Funds Overview Demo'[FundName]),
    MAX('Funds Overview Demo'[YTDReturn]),
    BLANK()
)
